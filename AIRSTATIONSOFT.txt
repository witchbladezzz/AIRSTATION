#define TINY_GSM_MODEM_SIM800
#define TINY_GSM_RX_BUFFER 1024

#include <Arduino.h>
#include <Wire.h>
#include <WiFi.h>
#include <WiFiManager.h>
#include <PubSubClient.h>
#include <TinyGsmClient.h>
#include <Adafruit_AHTX0.h>
#include <DFRobot_ENS160.h>
#include <TinyGPSPlus.h>

// =====================================================
// RIGHTECH MQTT
// =====================================================
const char* MQTT_HOST = "dev.rightech.io";
const int MQTT_PORT = 1883;

const char* MQTT_CLIENT_ID =
"mqtt-sibagatullin_roman74-q5st86";

// =====================================================
// BEELINE APN
// =====================================================
const char APN[]  = "internet.beeline.ru";
const char USER[] = "beeline";
const char PASS[] = "beeline";

// =====================================================
// WIFI MANAGER
// =====================================================
const char* WIFI_AP_NAME = "AirStation-Setup";
const char* WIFI_AP_PASS = "12345678";

const bool RESET_WIFI_EACH_BOOT = true;

// =====================================================
// I2C
// =====================================================
#define SDA_PIN 8
#define SCL_PIN 9

// =====================================================
// PMS7003
// PMS7003 TX -> GPIO4
// =====================================================
#define PMS_RX 4
HardwareSerial PMS(0);

// =====================================================
// GPS
// GPS TX -> GPIO6
// =====================================================
#define GPS_RX 6
HardwareSerial GPS(1);
TinyGPSPlus gps;

// =====================================================
// SIM800L
// SIM800L TX -> GPIO2
// SIM800L RX -> GPIO3 через 2 кОм
// =====================================================
#define SIM_RX 2
#define SIM_TX 3
HardwareSerial SIM(2);

// =====================================================
// SENSORS
// =====================================================
Adafruit_AHTX0 aht;
DFRobot_ENS160_I2C ENS160(&Wire, 0x53);

// =====================================================
// WIFI MQTT
// =====================================================
WiFiClient wifiClient;
PubSubClient wifiMqtt(wifiClient);

// =====================================================
// GSM MQTT
// =====================================================
TinyGsm modem(SIM);
TinyGsmClient gsmClient(modem);
PubSubClient gsmMqtt(gsmClient);

// =====================================================
// FLAGS
// =====================================================
bool useGsm = false;
bool wifiLocked = false;

// =====================================================
// PMS DATA
// =====================================================
struct PMSData {
  uint16_t pm1_0;
  uint16_t pm2_5;
  uint16_t pm10;
};

PMSData pms;

// =====================================================
// PMS7003 READ
// =====================================================
bool readPMS(PMSData &d) {

  while (PMS.available() >= 32) {

    if (PMS.read() == 0x42 && PMS.read() == 0x4D) {

      uint8_t b[30];

      PMS.readBytes(b, 30);

      d.pm1_0 = (b[4] << 8) | b[5];
      d.pm2_5 = (b[6] << 8) | b[7];
      d.pm10  = (b[8] << 8) | b[9];

      return true;
    }
  }

  return false;
}

// =====================================================
// WIFI MANAGER CONNECT
// =====================================================
void connectWiFiManager() {

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("WiFi already connected");
    return;
  }

  Serial.println();
  Serial.println("===== WIFI MANAGER =====");

  WiFi.mode(WIFI_STA);
  WiFi.setAutoReconnect(true);

  WiFiManager wm;

  static bool firstRun = true;

  if (RESET_WIFI_EACH_BOOT && firstRun) {
    Serial.println("Reset WiFi settings for this boot");
    wm.erase();
    firstRun = false;
  }

  wm.setConfigPortalTimeout(180);

  bool ok = wm.autoConnect(
    WIFI_AP_NAME,
    WIFI_AP_PASS
  );

  if (!ok) {
    Serial.println("WiFi config failed, restart...");
    delay(3000);
    ESP.restart();
  }

  Serial.print("WiFi OK IP=");
  Serial.println(WiFi.localIP());

  wifiLocked = true;
  useGsm = false;
}

// =====================================================
// WIFI MQTT CONNECT
// =====================================================
void connectWifiMqtt() {

  wifiMqtt.setServer(MQTT_HOST, MQTT_PORT);
  wifiMqtt.setKeepAlive(60);
  wifiMqtt.setBufferSize(512);

  while (!wifiMqtt.connected()) {

    Serial.print("WiFi MQTT connect... ");

    if (wifiMqtt.connect(MQTT_CLIENT_ID)) {

      Serial.println("OK");

    } else {

      Serial.print("FAIL state=");
      Serial.println(wifiMqtt.state());

      delay(3000);
    }
  }
}

// =====================================================
// GSM MQTT ATTEMPT
// =====================================================
bool tryGsmMqtt() {

  if (wifiLocked) {
    return false;
  }

  Serial.println();
  Serial.println("===== TRY GSM MQTT =====");

  SIM.begin(
    9600,
    SERIAL_8N1,
    SIM_RX,
    SIM_TX
  );

  delay(3000);

  Serial.println("MODEM TEST...");

  if (!modem.testAT()) {
    Serial.println("GSM FAIL: modem no AT response");
    return false;
  }

  Serial.println("MODEM OK");

  Serial.print("MODEM: ");
  Serial.println(modem.getModemInfo());

  Serial.println("WAIT GSM NETWORK...");

  if (!modem.waitForNetwork(30000)) {
    Serial.println("GSM FAIL: network");
    return false;
  }

  Serial.println("GSM NETWORK OK");

  Serial.print("CSQ: ");
  Serial.println(modem.getSignalQuality());

  Serial.print("Operator: ");
  Serial.println(modem.getOperator());

  Serial.println("CONNECT GPRS...");

  if (!modem.gprsConnect(APN, USER, PASS)) {
    Serial.println("GSM FAIL: GPRS");
    return false;
  }

  Serial.println("GPRS OK");

  Serial.print("IP: ");
  Serial.println(modem.localIP());

  gsmMqtt.setServer(MQTT_HOST, MQTT_PORT);
  gsmMqtt.setKeepAlive(60);
  gsmMqtt.setSocketTimeout(30);
  gsmMqtt.setBufferSize(512);

  Serial.println("GSM MQTT CONNECT...");

  for (int i = 1; i <= 3; i++) {

    Serial.print("GSM MQTT try ");
    Serial.print(i);
    Serial.print("... ");

    if (gsmMqtt.connect(MQTT_CLIENT_ID)) {

      Serial.println("OK");

      useGsm = true;
      wifiLocked = false;

      return true;

    } else {

      Serial.print("FAIL state=");
      Serial.println(gsmMqtt.state());

      delay(3000);
    }
  }

  Serial.println("GSM MQTT FAILED");
  return false;
}

// =====================================================
// SWITCH TO WIFI
// =====================================================
void switchToWiFi() {

  Serial.println();
  Serial.println("===== FALLBACK TO WIFI =====");

  useGsm = false;
  wifiLocked = true;

  if (gsmMqtt.connected()) {
    gsmMqtt.disconnect();
  }

  modem.gprsDisconnect();

  connectWiFiManager();
  connectWifiMqtt();
}

// =====================================================
// ENSURE CONNECTION
// =====================================================
void ensureConnection() {

  if (wifiLocked) {

    if (WiFi.status() != WL_CONNECTED) {
      Serial.println("WiFi lost, reconnect...");
      connectWiFiManager();
    }

    if (!wifiMqtt.connected()) {
      connectWifiMqtt();
    }

    wifiMqtt.loop();
    return;
  }

  if (useGsm) {

    if (!gsmMqtt.connected()) {

      Serial.println("GSM MQTT lost, trying reconnect...");

      if (!tryGsmMqtt()) {
        switchToWiFi();
      }
    }

    gsmMqtt.loop();
    return;
  }
}

// =====================================================
// MQTT PUBLISH UNIVERSAL
// =====================================================
bool publishMqtt(const char* topic, const char* payload) {

  bool ok = false;

  if (useGsm && gsmMqtt.connected()) {

    ok = gsmMqtt.publish(topic, payload);

    Serial.print("[GSM] ");

  } else {

    ok = wifiMqtt.publish(topic, payload);

    Serial.print("[WiFi] ");
  }

  Serial.print(topic);
  Serial.print(" = ");
  Serial.print(payload);
  Serial.print(" : ");
  Serial.println(ok ? "OK" : "FAIL");

  delay(200);

  return ok;
}

bool sendFloat(const char* topic, float v) {

  char payload[20];

  dtostrf(v, 1, 2, payload);

  return publishMqtt(topic, payload);
}

bool sendInt(const char* topic, int v) {

  char payload[20];

  sprintf(payload, "%d", v);

  return publishMqtt(topic, payload);
}

// =====================================================
// SEND ALL DATA
// =====================================================
bool sendAllData(
  float temperature,
  float hum,
  uint16_t eco2,
  uint16_t tvoc,
  bool pmsOk
) {

  bool ok = true;

  ok &= sendFloat(
    "base/state/Temperature",
    temperature
  );

  ok &= sendFloat(
    "base/state/Humidity",
    hum
  );

  ok &= sendInt(
    "base/state/eCO2",
    eco2
  );

  ok &= sendInt(
    "base/state/TVOC",
    tvoc
  );

  if (pmsOk) {

    ok &= sendInt(
      "base/state/PM1.0",
      pms.pm1_0
    );

    ok &= sendInt(
      "base/state/PM25",
      pms.pm2_5
    );

    ok &= sendInt(
      "base/state/PM10",
      pms.pm10
    );

  } else {

    Serial.println("PMS7003 NO DATA");
  }

  if (gps.location.isValid()) {

    char gpsPayload[80];

    sprintf(
      gpsPayload,
      "{\"lat\":%.6f,\"lon\":%.6f}",
      gps.location.lat(),
      gps.location.lng()
    );

    ok &= publishMqtt(
      "base/state/pos",
      gpsPayload
    );

    Serial.print("GPS=");
    Serial.println(gpsPayload);

  } else {

    Serial.println("GPS NO FIX");
  }

  return ok;
}

// =====================================================
// SETUP
// =====================================================
void setup() {

  Serial.begin(115200);
  delay(3000);

  Serial.println();
  Serial.println("==============================");
  Serial.println("AIR STATION START");
  Serial.println("GSM MQTT -> WIFI FALLBACK");
  Serial.println("==============================");

  // I2C
  Wire.begin(SDA_PIN, SCL_PIN);

  // PMS7003
  PMS.begin(
    9600,
    SERIAL_8N1,
    PMS_RX,
    -1
  );

  // GPS
  GPS.begin(
    9600,
    SERIAL_8N1,
    GPS_RX,
    -1
  );

  // AHT21
  if (!aht.begin()) {
    Serial.println("AHT21 FAIL");
  } else {
    Serial.println("AHT21 OK");
  }

  // ENS160
  while (NO_ERR != ENS160.begin()) {
    Serial.println("ENS160 FAIL");
    delay(1000);
  }

  Serial.println("ENS160 OK");

  ENS160.setPWRMode(
    ENS160_STANDARD_MODE
  );

  // =====================================================
  // FIRST TRY GSM MQTT
  // =====================================================
  if (tryGsmMqtt()) {

    Serial.println();
    Serial.println("GSM MQTT SELECTED");

  } else {

    Serial.println();
    Serial.println("GSM MQTT FAILED -> WIFI");

    switchToWiFi();
  }

  Serial.println();
  Serial.println("SYSTEM READY");
}

// =====================================================
// LOOP
// =====================================================
void loop() {

  ensureConnection();

  // GPS update
  while (GPS.available()) {
    gps.encode(GPS.read());
  }

  // AHT21
  sensors_event_t h, t;
  aht.getEvent(&h, &t);

  float temperature = t.temperature;
  float hum = h.relative_humidity;

  // ENS160
  ENS160.setTempAndHum(
    temperature,
    hum
  );

  uint16_t eco2 = ENS160.getECO2();
  uint16_t tvoc = ENS160.getTVOC();

  // PMS7003
  bool pmsOk = readPMS(pms);

  Serial.println();
  Serial.println("========== DATA ==========");

  Serial.print("Temperature: ");
  Serial.println(temperature);

  Serial.print("Humidity: ");
  Serial.println(hum);

  Serial.print("eCO2: ");
  Serial.println(eco2);

  Serial.print("TVOC: ");
  Serial.println(tvoc);

  if (pmsOk) {
    Serial.print("PM1.0: ");
    Serial.println(pms.pm1_0);

    Serial.print("PM2.5: ");
    Serial.println(pms.pm2_5);

    Serial.print("PM10: ");
    Serial.println(pms.pm10);
  }

  bool sent = sendAllData(
    temperature,
    hum,
    eco2,
    tvoc,
    pmsOk
  );

  // если вдруг GSM был выбран, но отправка не прошла — уходим на Wi-Fi
  if (!sent && useGsm) {

    Serial.println();
    Serial.println("GSM SEND FAILED -> SWITCH TO WIFI");

    switchToWiFi();

    sendAllData(
      temperature,
      hum,
      eco2,
      tvoc,
      pmsOk
    );
  }

  unsigned long start = millis();

  while (millis() - start < 10000) {

    if (useGsm && gsmMqtt.connected()) {
      gsmMqtt.loop();
    }

    if (wifiLocked && wifiMqtt.connected()) {
      wifiMqtt.loop();
    }

    delay(100);
  }
}
